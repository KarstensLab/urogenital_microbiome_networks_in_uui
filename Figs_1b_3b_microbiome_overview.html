#' Calculate given alpha diversity values
#'
#' Given a phyloseq-class object, calculates richness and returns a data frame
#' with those richness estimates for easier downstream analysis.
#'
#' Pielou can also be calculated. It's definition can be found here
#' https://www.slideshare.net/MarkMcGinley/introduction-to-biodiversity-8383743.
#' It is the ratio between Shannon diversity of the sample divided by the
#' maximum Shannon diversity for that sample. The maximum Shannon diversity
#' means all species represented equalled, while a Shannon diversity of 0 means
#' only one species exists. More information on Shannon diversity here
#' https://microbeatic.wordpress.com/2012/01/26/diversity-index-shannon-indexshannon-weaver-index-h/
#' 
calculate_alpha <- function(ps, richness_vals) {
  # Calculate all richness measured from phyloseq
  richness_info <-
    ps %>%
    phyloseq::estimate_richness(measures = richness_vals) %>%
    
    # The row names contain an index to map to cohort information
    tibble::rownames_to_column("sample_name") %>%
    
    # Join in the cohort information with sample name used in richness analysis
    dplyr::left_join(
      ps %>%
        phyloseq::sample_data() %>%
        data.frame() %>%
        dplyr::select(sample_name, cohort_sufu)
    )
  
  # Calculate Pielou if available, and separated because using separate package
  if ("Pielou" %in% richness_vals) {
    # Adjust diversity to calculate values because Pielou takes log of zero
    # abundance calculations so cannot compute, so performs minor adjustment to
    # calculate values
    ps_otu <- phyloseq::otu_table(ps)
    adj_ps <- ps
    ps_otu <- ps_otu + 0.01  # Add small adjustment
    otu_table(adj_ps) <- ps_otu
    
    pielou_vals <-
      adj_ps %>%
      microbiome::evenness(index = "pielou") %>%
      dplyr::rename(Pielou = pielou) %>%  # For consistency with others
      tibble::rownames_to_column("sample_name")
    
    # Save into previous created data frame
    richness_info <-
      richness_info %>%
      dplyr::left_join(pielou_vals, by = "sample_name")
    
    # Rearrange for niceness
    richness_info <-
      richness_info %>%
      dplyr::select(sample_name, cohort_sufu, everything())
  }
  
  richness_info
}


#' Create plot of alpha diversity
#'
#' @param df data.frame with columns `cohort_sufu` and any other diversity value
#' @param colval string of diversity value that exists in `df`
#'
#' @return ggplot2-object
#'
#' @example
#' plot_rich_stats(richness_vals_cohort, "Observed")
plot_rich_stats <- function(df, colval) {
  s_val <- sym(colval)
  df %>%
    mutate(cohort_sufu = factor(cohort_sufu,
                                levels = c("case", "control"))) %>%
    mutate(cohort_sufu = fct_recode(cohort_sufu,
                                    "UUI" = "case",
                                    "Control" = "control")) %>%
    select(sample_name, cohort_sufu, !!s_val) %>%
    ggboxplot(
      x = "cohort_sufu",
      y = colval,
     # add = "jitter",
      color = "cohort_sufu") +
    stat_compare_means(
      comparisons = list(c("Control", "UUI")),
      method = "wilcox.test") +
    xlab("Cohort") +
    theme(legend.position = "none")
}
